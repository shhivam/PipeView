    **2. Create `BandwidthMonitor/MenuBar/StatusBarController.swift`:**

    Create a `@MainActor` class `StatusBarController` that inherits from `NSObject` (required for `#selector` with `@objc` methods). It owns the NSStatusItem and NSMenu. It must be a class (not struct) because NSStatusItem must be strongly retained and the observation pattern uses `self` references.

    ```swift
    import AppKit
    import Observation
    import os

    @MainActor
    final class StatusBarController: NSObject {
        // MARK: - Properties

        private let statusItem: NSStatusItem
        private let networkMonitor: NetworkMonitor
        private let textBuilder = SpeedTextBuilder()

        // Hardcoded defaults for Phase 2 (per D-14: preferences deferred to Phase 5)
        // Per D-15: infrastructure supports all modes, but defaults are locked
        private let displayMode: DisplayMode = .auto
        private let unitMode: SpeedFormatter.UnitMode = .auto

        // MARK: - Init

        init(statusItem: NSStatusItem, networkMonitor: NetworkMonitor) {
            self.statusItem = statusItem
            self.networkMonitor = networkMonitor
            super.init()
        }

        // MARK: - Setup

        func setup() {
            // Set initial text to em dash (D-18 / UI-SPEC empty state: first launch shows em dash before first poll)
            updateStatusItemText(text: "\u{2014}")

            // Build and attach the menu (D-13: both left-click and right-click open same NSMenu)
            statusItem.menu = buildMenu()

            // Start observing NetworkMonitor state changes
            startObserving()

            Logger.menuBar.info("StatusBarController setup complete")
        }

        // MARK: - Observation (Pattern 3 from RESEARCH.md)

        // withObservationTracking fires only ONCE per registration.
        // Must re-register in onChange callback for continuous updates.
        // IMPORTANT: Do NOT read properties in onChange (it fires during willSet, values are OLD).
        // Read properties in the apply closure, which runs when startObserving() is called.
        private func startObserving() {
            withObservationTracking {
                // Access the properties to observe -- this registers tracking
                let speed = self.networkMonitor.aggregateSpeed
                let hasInterfaces = !self.networkMonitor.interfaceSpeeds.isEmpty

                // Build display text using SpeedTextBuilder
                let text = self.textBuilder.build(
                    speed: speed,
                    mode: self.displayMode,
                    unit: self.unitMode,
                    hasInterfaces: hasInterfaces
                )

                self.updateStatusItemText(text: text)
            } onChange: {
                // Re-register on next run loop tick (after didSet completes)
                Task { @MainActor [weak self] in
                    self?.startObserving()
                }
            }
        }

        // MARK: - Status Item Text

        // Per D-04: Use monospacedDigitSystemFont for tabular figures to prevent jitter
        // Per BAR-04: Fixed-width formatting via NSAttributedString
        private func updateStatusItemText(text: String) {
            let font = NSFont.monospacedDigitSystemFont(
                ofSize: NSFont.systemFontSize,
                weight: .regular
            )
            statusItem.button?.attributedTitle = NSAttributedString(
                string: text,
                attributes: [.font: font]
            )
        }

        // MARK: - Menu (D-12, D-13)

        // Per D-12: Menu items in order: Metrics (disabled), Preferences (disabled),
        //           separator, About Bandwidth Monitor, Quit Bandwidth Monitor
        // Per D-13: Both left-click and right-click open the same NSMenu
        //           (achieved by setting statusItem.menu, which handles both click types)
        private func buildMenu() -> NSMenu {
            let menu = NSMenu()

            // Item 1: Metrics (disabled, coming in Phase 4)
            let metrics = NSMenuItem(title: "Metrics", action: nil, keyEquivalent: "")
            metrics.isEnabled = false
            menu.addItem(metrics)

            // Item 2: Preferences (disabled, coming in Phase 5)
            let prefs = NSMenuItem(title: "Preferences", action: nil, keyEquivalent: "")
            prefs.isEnabled = false
            menu.addItem(prefs)

            // Item 3: Separator
            menu.addItem(.separator())

            // Item 4: About Bandwidth Monitor
            let about = NSMenuItem(
                title: "About Bandwidth Monitor",
                action: #selector(showAbout),
                keyEquivalent: ""
            )
            about.target = self
            menu.addItem(about)

            // Item 5: Quit Bandwidth Monitor (Cmd+Q shortcut per UI-SPEC)
            let quit = NSMenuItem(
                title: "Quit Bandwidth Monitor",
                action: #selector(NSApplication.terminate(_:)),
                keyEquivalent: "q"
            )
            menu.addItem(quit)

            return menu
        }

        // MARK: - Actions

        // About panel: use system standard About panel (reads Info.plist automatically)
        // Per RESEARCH.md: activate first so panel is visible (agent apps can be behind other windows)
        @objc private func showAbout() {
            NSApplication.shared.activate(ignoringOtherApps: true)
            NSApplication.shared.orderFrontStandardAboutPanel(options: [:])
        }
    }
    ```

    **1. Create `BandwidthMonitor/AppDelegate.swift`:**

    ```swift
    import AppKit
    import ServiceManagement
    import os

    @MainActor
    class AppDelegate: NSObject, NSApplicationDelegate {
        // MARK: - Properties

        // Strong references prevent deallocation (Pitfall 1 from RESEARCH.md)
        private var statusItem: NSStatusItem!
        private var statusBarController: StatusBarController?
        private let networkMonitor = NetworkMonitor()

        // MARK: - NSApplicationDelegate

        func applicationDidFinishLaunching(_ notification: Notification) {
            // Per D-17: First launch goes straight to menu bar, no onboarding, no splash

            // 1. Create the status item (variable length for speed text)
            statusItem = NSStatusBar.system.statusItem(
                withLength: NSStatusItem.variableLength
            )

            // 2. Create and setup the status bar controller
            statusBarController = StatusBarController(
                statusItem: statusItem,
                networkMonitor: networkMonitor
            )
            statusBarController?.setup()

            // 3. Start the monitoring engine
            networkMonitor.start()

            // 4. Register as login item (D-16: auto-register on first launch, no prompt)
            registerLoginItem()

            Logger.lifecycle.info("Application launched, menu bar active")
        }

        func applicationWillTerminate(_ notification: Notification) {
            networkMonitor.stop()
            Logger.lifecycle.info("Application terminating")
        }

        // MARK: - Login Item (SYS-01, D-16)

        // Per D-16: Auto-register as login item on first launch, no user prompt.
        // Per RESEARCH.md Pitfall 4: May not work in debug builds from Xcode -- expected behavior.
        // Per RESEARCH.md: Do NOT cache status in UserDefaults. SMAppService.mainApp.status is source of truth.
        private func registerLoginItem() {
            do {
                if SMAppService.mainApp.status != .enabled {
                    try SMAppService.mainApp.register()
                    Logger.lifecycle.info("Registered as login item successfully")
                } else {
                    Logger.lifecycle.debug("Already registered as login item")
                }
            } catch {
                // Discretion: fail silently for user, log for debugging
                // This is expected to fail in Xcode debug builds (not running from /Applications)
                Logger.lifecycle.error("Failed to register login item: \(error.localizedDescription)")
            }
        }
    }
    ```
