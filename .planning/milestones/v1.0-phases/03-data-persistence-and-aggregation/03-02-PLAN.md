Create `BandwidthMonitor/Persistence/BandwidthRecorder.swift`:

```swift
import Foundation
import GRDB
import Observation
import os

@MainActor
final class BandwidthRecorder {
    // MARK: - Configuration

    /// Number of snapshots to accumulate before writing (per D-01: 5 cycles * 2s = 10s)
    let accumulationCount: Int

    /// Assumed polling interval for converting speed to bytes
    let pollingInterval: TimeInterval

    // MARK: - Dependencies

    private let networkMonitor: NetworkMonitor
    private let database: AppDatabase
    private let logger = Logger.persistence

    // MARK: - State

    private var buffer: [NetworkSnapshot] = []
    private var recordingTask: Task<Void, Never>?
    private(set) var isRecording: Bool = false
    private(set) var totalSamplesWritten: Int = 0

    // MARK: - Init

    init(
        networkMonitor: NetworkMonitor,
        database: AppDatabase,
        accumulationCount: Int = 5,
        pollingInterval: TimeInterval = 2.0
    ) {
        self.networkMonitor = networkMonitor
        self.database = database
        self.accumulationCount = accumulationCount
        self.pollingInterval = pollingInterval
    }

    // MARK: - Lifecycle

    func start() {
        guard !isRecording else { return }
        isRecording = true
        buffer.removeAll()
        logger.info("BandwidthRecorder started (accumulating \(self.accumulationCount) snapshots)")

        recordingTask = Task { [weak self] in
            guard let self else { return }

            // Use withObservationTracking + AsyncStream to observe latestSnapshot changes
            // Same pattern as StatusBarController but yielding into an async stream
            let snapshots = AsyncStream<NetworkSnapshot> { continuation in
                @Sendable func observe() {
                    let snapshot = withObservationTracking {
                        self.networkMonitor.latestSnapshot
                    } onChange: {
                        Task { @MainActor in
                            observe()
                        }
                    }
                    continuation.yield(snapshot)
                }
                observe()

                continuation.onTermination = { _ in
                    // Stream cancelled
                }
            }

            for await snapshot in snapshots {
                guard !Task.isCancelled else { break }

                // Skip empty snapshots (no active interfaces)
                guard !snapshot.interfaceSpeeds.isEmpty else { continue }

                self.buffer.append(snapshot)

                if self.buffer.count >= self.accumulationCount {
                    let samples = self.buffer
                    self.buffer.removeAll()
                    await self.writeSamples(from: samples)
                }
            }
        }
    }

    func stop() {
        recordingTask?.cancel()
        recordingTask = nil
        isRecording = false
        logger.info("BandwidthRecorder stopped (total samples written: \(self.totalSamplesWritten))")
    }

    /// Flush any buffered snapshots to the database (called on app termination per Pitfall 5).
    /// Writes whatever is in the buffer even if < accumulationCount snapshots.
    func flush() async {
        guard !buffer.isEmpty else { return }
        let samples = buffer
        buffer.removeAll()
        logger.info("Flushing \(samples.count) buffered snapshots")
        await writeSamples(from: samples)
    }

    // MARK: - Writing

    /// Convert accumulated snapshots to RawSample records and write to database.
    /// Called off main thread to avoid blocking UI (per Pitfall 1).
    private nonisolated func writeSamples(from snapshots: [NetworkSnapshot]) async {
        let pollingInterval = await self.pollingInterval
        let count = snapshots.count
        let duration = pollingInterval * Double(count)
        let timestamp = snapshots.last!.timestamp.timeIntervalSince1970

        // Group speeds by interface across all snapshots
        var interfaceSpeeds: [String: (totalBytesIn: Double, totalBytesOut: Double)] = [:]
        for snapshot in snapshots {
            for ifSpeed in snapshot.interfaceSpeeds {
                let key = ifSpeed.interface.bsdName
                let existing = interfaceSpeeds[key, default: (totalBytesIn: 0, totalBytesOut: 0)]
                // Per D-11: Convert speed (bytes/sec) to total bytes for this poll interval
                interfaceSpeeds[key] = (
                    totalBytesIn: existing.totalBytesIn + ifSpeed.speed.bytesInPerSecond * pollingInterval,
                    totalBytesOut: existing.totalBytesOut + ifSpeed.speed.bytesOutPerSecond * pollingInterval
                )
            }
        }

        // Create one RawSample per interface
        var rawSamples: [RawSample] = []
        for (interfaceId, totals) in interfaceSpeeds {
            rawSamples.append(RawSample(
                id: nil,
                interfaceId: interfaceId,
                timestamp: timestamp,
                bytesIn: totals.totalBytesIn,
                bytesOut: totals.totalBytesOut,
                duration: duration
            ))
        }

        guard !rawSamples.isEmpty else { return }

        do {
            try await database.dbWriter.write { db in
                for var sample in rawSamples {
                    try sample.insert(db)
                }
            }
            await MainActor.run {
                self.totalSamplesWritten += rawSamples.count
                self.logger.debug("Wrote \(rawSamples.count) raw samples (duration=\(duration)s)")
            }
        } catch {
            await MainActor.run {
                self.logger.error("Failed to write raw samples: \(error.localizedDescription)")
            }
        }
    }
}
```

**Important design notes:**
- `writeSamples` is `nonisolated` so the actual `dbWriter.write` call executes off the main thread (per Pitfall 1 from research)
- `flush()` is `async` because it calls `writeSamples` which does async database work
- The buffer is always cleared before writing (not after) to prevent data loss if write fails
- Empty snapshots (no interfaces active) are skipped to avoid empty writes
- Uses `database.dbWriter` (not `database.pool`) because AppDatabase stores `any DatabaseWriter`
